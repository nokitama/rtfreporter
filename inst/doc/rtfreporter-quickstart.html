rtfreporter — Clinical RTF Reporting

Overview

rtfreporter produces clinical-trial–style RTF reports (tables, listings, figures) without depending on r2rtf or any other RTF library.

Document hierarchy: Document → Section → Page → Block Each section carries its own header / footer. When you omit them, the previous section’s values are inherited automatically.

1 Minimal one-page report

report <- rtfreport$new()

sec <- report$add_section(
  header = rtf_header(rows = list(
    c(l = "Protocol: RTF-101",          r = "ACME Pharma"),
    c(l = "Table 14.1.1 Demographics",  r = "Page {AUTO_PAGE} of {AUTO_TOTAL_PAGES}")
  )),
  footer = rtf_footer(rows = list(
    c(l = "Source: DM domain.", r = "CONFIDENTIAL")
  ))
)

df <- data.frame(
  Subject = c("001", "002", "003"),
  Arm     = c("Active", "Placebo", "Active"),
  Age     = c(34L, 45L, 28L),
  Sex     = c("M", "F", "M"),
  stringsAsFactors = FALSE
)

report$add_page(
  section_index = sec,
  content = list(
    list(type = "table",
         data = rtftable$new(df,
           col_rel_width    = c(2, 2, 1, 1),
           row_height_twips = 280L))
  ),
  footer_notes = "Source: DM domain. One row per subject."
)

generate_rtfreport(report, tempfile(fileext = ".rtf"), overwrite = TRUE)

2 Lab shift table — one section per analyte

The shift table is the canonical multi-section clinical TFL. Each analyte (ALT, AST, HGB, …) occupies its own RTF section so the analyte name appears in the section header bottom-left, changing from page to page.

RTF headers and footers are per-section; this is the correct way to vary header content across pages of the same document.

library(rtfreporter)

# --- synthetic shift-count data ----------------------------------------------
make_shift_df <- function(a_low, a_norm, a_high, b_low, b_norm, b_high) {
  data.frame(
    Baseline = c("Low", "Normal", "High"),
    A_Low    = a_low,  A_Normal = a_norm,  A_High = a_high,
    B_Low    = b_low,  B_Normal = b_norm,  B_High = b_high,
    stringsAsFactors = FALSE, check.names = FALSE
  )
}

lab_data <- list(
  list(label = "ALT (Alanine Aminotransferase)",
       df    = make_shift_df(c(4,0,0), c(1,13,1), c(0,2,3),
                             c(3,0,0), c(1,14,0), c(0,1,4))),
  list(label = "AST (Aspartate Aminotransferase)",
       df    = make_shift_df(c(5,0,0), c(0,11,2), c(0,1,5),
                             c(4,0,0), c(1,12,1), c(0,2,4))),
  list(label = "HGB (Hemoglobin)",
       df    = make_shift_df(c(2,1,0), c(1,16,1), c(0,1,2),
                             c(3,0,0), c(0,15,2), c(0,2,2)))
)

# --- shared table structure --------------------------------------------------
spanning_hdr <- list(
  list(from = 2L, to = 4L, label = "Treatment A  (N=24)", underline = TRUE),
  list(from = 5L, to = 7L, label = "Treatment B  (N=24)", underline = TRUE)
)
col_hdr    <- c("Baseline", "Low", "Normal", "High", "Low", "Normal", "High")
col_widths <- c(2160L, 900L, 900L, 900L, 900L, 900L, 900L)

col_spec_shift <- lapply(seq_len(7L), function(j)
  list(col = j, align = if (j == 1L) "left" else "center"))

make_shift_table <- function(df) {
  rtftable$new(
    data                = df,
    col_header          = col_hdr,
    spanning_header     = spanning_hdr,
    col_spec            = col_spec_shift,
    column_widths_twips = col_widths,
    border              = "tfl",
    row_height_twips    = 280L
  )
}

# --- common footer (same for all sections) -----------------------------------
common_footer <- rtf_footer(
  rows = list(
    c(l = paste0("ALT=Alanine Aminotransferase; ",
                 "AST=Aspartate Aminotransferase; HGB=Hemoglobin")),
    c(l = "Low/Normal/High: site reference range categories.  n = number of subjects.",
      r = "CONFIDENTIAL")
  )
)

# --- build report: one section per analyte -----------------------------------
report <- rtfreport$new()

for (item in lab_data) {
  sec_header <- rtf_header(
    rows = list(
      c(l = "Protocol: LAB-001",                           r = "ACME Pharma"),
      c(l = "Study Title: Phase III Safety Lab Assessment", r = "Page {AUTO_PAGE} of {AUTO_TOTAL_PAGES}"),
      c(c = "Table 14.3.1  Laboratory Shift Table (Safety Population)"),
      c(l = item$label)       # ← analyte name: bottom-left, unique per section
    )
  )

  sec <- report$add_section(header = sec_header, footer = common_footer)
  report$add_page(
    section_index = sec,
    content       = list(list(type = "table", data = make_shift_table(item$df))),
    footer_notes  = paste0("Source: LB domain.  ",
      "Subjects with both a baseline and at least one post-baseline assessment included.")
  )
}

generate_rtfreport(report, tempfile(fileext = ".rtf"), overwrite = TRUE)

The resulting file has 3 sections (one per analyte), each with its own header row 4 set to the analyte label.

3 Embedding figures (ggplot2)

rtfplot embeds a PNG or JPEG image binary directly into the RTF \pict block. The default alignment is center; specify width_twips to control the display size.

library(ggplot2)

# --- build a ggplot and save to a temp PNG -----------------------------------
p <- ggplot(
  data.frame(x = 1:10, y = (1:10)^2 + rnorm(10, sd = 3)),
  aes(x = x, y = y)) +
  geom_point(size = 3, colour = "#2166AC") +
  geom_smooth(method = "lm", se = FALSE, colour = "#D6604D") +
  labs(title = "Lab value vs. Visit (simulated)", x = "Visit", y = "Lab value") +
  theme_bw(base_size = 11)

png_path <- tempfile(fileext = ".png")
ggsave(png_path, plot = p, width = 7, height = 4.5, dpi = 150)

# --- embed in RTF ------------------------------------------------------------
plot_obj <- rtfplot$new(
  path        = png_path,
  width_twips = 9000L   # ~6.25 in; height computed from aspect ratio
)

report <- rtfreport$new()
sec <- report$add_section(
  header = rtf_header(rows = list(
    c(l = "Protocol: FIG-001",           r = "ACME Pharma"),
    c(l = "Figure 14.1.1  Lab Value vs. Visit (Safety Population)")
  )),
  footer = rtf_footer(rows = list(c(l = "Source: LB domain.")))
)
report$add_page(sec, content = list(list(type = "figure", data = plot_obj)))

generate_rtfreport(report, tempfile(fileext = ".rtf"), overwrite = TRUE)

4 Figure and table on the same page

Keep the figure width moderate so both elements fit on one page.

tbl_summary <- rtftable$new(
  data.frame(
    Statistic = c("n", "Mean", "SD"),
    Value     = c("30", "42.7", "8.4"),
    stringsAsFactors = FALSE
  ),
  table_width_pct  = 40,     # 40 % of writable width
  table_align      = "left",
  row_height_twips = 280L
)

report2 <- rtfreport$new()
sec2 <- report2$add_section(
  header = rtf_header(rows = list(
    c(l = "Protocol: FIG-001", r = "ACME Pharma"),
    c(l = "Figure + Summary Table")
  )),
  footer = rtf_footer(rows = list(c(l = "Source: LB domain.", r = "CONFIDENTIAL")))
)
report2$add_page(
  section_index = sec2,
  content = list(
    list(type = "figure", data = rtfplot$new(png_path, width_twips = 7200L)),
    list(type = "table",  data = tbl_summary)
  )
)
generate_rtfreport(report2, tempfile(fileext = ".rtf"), overwrite = TRUE)

5 Table width and alignment

Parameter Description
table_width_pct = 70 70 % of writable page width
table_width_twips = 9000L Absolute: 9000 twips = 6.25 in
table_align = "center" Center the table on the page
table_align = "right" Right-align the table 
df <- data.frame(A = 1:3, B = letters[1:3], C = c(1.1, 2.2, 3.3))

# 70 % of writable width, left-aligned (default)
tbl_left <- rtftable$new(df, table_width_pct = 70)

# Absolute width, centered
tbl_center <- rtftable$new(df,
  table_width_twips = 9000L,
  table_align       = "center")

# Absolute width, right-aligned, with relative column widths 3:1:1
tbl_right <- rtftable$new(df,
  table_width_twips = 7200L,
  table_align       = "right",
  col_rel_width     = c(3, 1, 1))

# Full writable width (100 %)
tbl_full  <- rtftable$new(df, table_width_pct = 100)

6 Column width control

Three ways to set column widths, in order of precedence:

Method Parameter Notes
Absolute column_widths_twips = c(2880L, 1440L, 1440L) One value per column
Relative col_rel_width = c(2, 1, 1) + table_width_twips Proportional split
Auto auto_col_widths(df) Font-aware estimation 
# Absolute column widths (twips)
tbl_abs <- rtftable$new(df,
  column_widths_twips = c(4320L, 2160L, 2160L))

# Relative widths 2:1:1 within a 7-inch table
tbl_rel <- rtftable$new(df,
  col_rel_width     = c(2, 1, 1),
  table_width_twips = 10080L)

# Auto-sizing: widths computed from the widest cell in each column
w <- auto_col_widths(df, table_width_twips = 14400L)
tbl_auto <- rtftable$new(df, column_widths_twips = w)

7 Header / footer layout

Keys l, c, r map to left, center, right alignment. An unnamed single element defaults to center (for both header and footer).

# Single row, two columns
sec_hf1 <- rtf_header(rows = list(
  c(l = "Protocol: RTF-101", r = "Page {AUTO_PAGE} of {AUTO_TOTAL_PAGES}")
))

# Multi-row header
sec_hf2 <- rtf_header(rows = list(
  c(l = "Protocol: RTF-101",    r = "ACME Pharma"),
  c(l = "Phase III Safety Study", r = "Page {AUTO_PAGE} of {AUTO_TOTAL_PAGES}"),
  c(c = "Table 14.3.1  Shift Table (Safety Population)"),
  c(l = "ALT (Alanine Aminotransferase)")   # bottom-left: analyte name
))

# Footer with two rows; top border on row 1 only (default)
sec_ftr <- rtf_footer(rows = list(
  c(l = "ALT=Alanine Aminotransferase",     r = "CONFIDENTIAL"),
  c(l = "Low/Normal/High: site reference range categories.")
))

# Single unnamed row → center (default)
ftr_center <- rtf_footer(rows = list(c("Confidential — Internal Use Only")))

8 Page tokens

Token Behaviour
{AUTO_PAGE} Dynamic page number (\chpgn), updated per page by the viewer
{AUTO_TOTAL_PAGES} Dynamic NUMPAGES field, updated by the viewer — recommended for assembled docs
{TOTAL_PAGES} Static total computed at render time by generate_rtfreport()
{SECTION_PAGES} Dynamic SECTIONPAGES field (pages in the current section)

Recommended: "Page {AUTO_PAGE} of {AUTO_TOTAL_PAGES}".

Use {TOTAL_PAGES} only when generating a single, self-contained file where the total page count is known at render time.

10 Multi data.frame table

When data is a list of data.frames each is rendered in sequence, with its own column-header rows. This lets you combine cohorts or time points in one table without duplicating the spanning header.

df_a <- data.frame(Label = c("Low", "Normal", "High"), n = c(4L, 13L, 3L))
df_b <- data.frame(Label = c("Low", "Normal", "High"), n = c(3L, 14L, 4L))

tbl_multi <- rtftable$new(
  data       = list(df_a, df_b),
  col_header = list(
    c("Baseline", "Arm A"),   # header for df_a
    c("Baseline", "Arm B")    # header for df_b
  ),
  column_widths_twips = c(3600L, 2160L),
  border              = "tfl",
  row_height_twips    = 280L
)

11 Border customization

# TFL clinical preset (default): header top+bottom, last-row bottom
tbl_tfl <- rtftable$new(df, border = "tfl")

# Custom per-zone
tbl_custom <- rtftable$new(df,
  border = rtf_table_border(
    header   = rtf_border(top = rtf_border_side(), bottom = rtf_border_side()),
    last_row = rtf_border(bottom = rtf_border_side("double", 10L))
  )
)

# Header with bottom divider, footer with thick blue top border
hdr_bordered <- rtf_header(
  rows   = list(c(c = "Study Title")),
  border = rtf_border(bottom = rtf_border_side("single"))
)
ftr_blue <- rtf_footer(
  rows   = list(c(l = "Confidential")),
  border = rtf_border_top(style = "thick", width = 20L, color = "#003366")
)

12 Assembling multiple RTF files

assemble_rtf() joins independently generated RTF files into one document. Use dynamic tokens ({AUTO_TOTAL_PAGES}) so page numbers are correct across the assembled output.

make_rtf <- function(df, hdr_text, path) {
  r <- rtfreport$new()
  sec <- r$add_section(
    header = c(l = hdr_text, r = "Page {AUTO_PAGE} of {AUTO_TOTAL_PAGES}"),
    footer = c(l = "CONFIDENTIAL")
  )
  r$add_page(sec, content = list(list(type = "table", data = df)))
  generate_rtfreport(r, path, overwrite = TRUE)
}

f1 <- tempfile(fileext = ".rtf"); make_rtf(mtcars[1:3, ], "Table 14.1 DM", f1)
f2 <- tempfile(fileext = ".rtf"); make_rtf(mtcars[4:6, ], "Table 14.2 AE", f2)

assembled <- tempfile(fileext = ".rtf")
assemble_rtf(c(f1, f2), assembled, overwrite = TRUE)

13 Auto column widths

auto_col_widths() estimates widths from the actual content and scales them to a target table width.

df_ae <- data.frame(
  USUBJID  = c("001-001", "001-002"),
  AETERM   = c("Headache", "Nausea"),
  AESTDTC  = c("2024-01-15", "2024-02-03"),
  AESER    = c("N", "N"),
  stringsAsFactors = FALSE
)

# Scale to full writable width (14400 twips = 10 in landscape letter)
w <- auto_col_widths(df_ae,
  col_header        = "Subject ID | AE Term | Start Date | Serious",
  table_width_twips = 14400L)

tbl_ae <- rtftable$new(df_ae,
  col_header          = c("Subject ID", "AE Term", "Start Date", "Serious"),
  column_widths_twips = w,
  border              = "tfl",
  row_height_twips    = 280L)

Quick-reference

rtftable parameters

Parameter Type Purpose
data data.frame or list Content rows
col_header chr / list Column header text
spanning_header list Spanning (grouped) header
col_spec list Per-column align / bold / italic / indent
border "tfl" / object Cell borders
col_rel_width numeric vector Proportional column widths
column_widths_twips integer vector Absolute column widths
table_width_twips integer Table total width (twips)
table_width_pct numeric 0–100 Table width as % of page
table_align "left" / "center" / "right" Table position on page
row_height_twips integer Data row height
blank_rows integer vector Insert blank separator rows

rtfplot parameters

Parameter Default Purpose
path — PNG or JPEG file path
width_twips NULL (full width) Display width
height_twips NULL (aspect ratio) Display height
align "center" "left" / "center" / "right"

Header / footer single-row defaults

Keys used Columns Alignments
l= only 1 left
r= only 1 right
c= only 1 center
l= + r= 2 left, right
unnamed single value 1 center (default)
l= + c= + r= 3 left, center, right
